// Traffic Light Control using Arduino UNO

const int LED1 = 2;   // Green 1
const int LED2 = 3;   // Yellow 1
const int LED3 = 4;   // Red 1
const int LED4 = 5;   // Green 2
const int LED5 = 6;   // Red 2
const int BUTTON = 7; // Push Button

void setup()
{
    pinMode(LED1, OUTPUT);
    pinMode(LED2, OUTPUT);
    pinMode(LED3, OUTPUT);
    pinMode(LED4, OUTPUT);
    pinMode(LED5, OUTPUT);

    pinMode(BUTTON, INPUT_PULLUP);
}

void loop()
{
    // If button is pressed
    if (digitalRead(BUTTON) == LOW)
    {
        // Road 1 Green, Road 2 Red
        digitalWrite(LED1, HIGH);
        digitalWrite(LED2, LOW);
        digitalWrite(LED3, LOW);

        digitalWrite(LED4, LOW);
        digitalWrite(LED5, HIGH);

        delay(5000);

        // Road 1 Yellow
        digitalWrite(LED1, LOW);
        digitalWrite(LED2, HIGH);
        delay(2000);

        digitalWrite(LED2, LOW);
        digitalWrite(LED3, HIGH);

        // Road 2 Green
        digitalWrite(LED5, LOW);
        digitalWrite(LED4, HIGH);

        delay(5000);

        // Road 2 Red
        digitalWrite(LED4, LOW);
        digitalWrite(LED5, HIGH);

        delay(1000);

        // Return to initial state
        digitalWrite(LED3, LOW);
        digitalWrite(LED1, HIGH);
    }
    else
    {
        // Default state
        digitalWrite(LED1, HIGH);
        digitalWrite(LED2, LOW);
        digitalWrite(LED3, LOW);

        digitalWrite(LED4, LOW);
        digitalWrite(LED5, HIGH);
    }
}
